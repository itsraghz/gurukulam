                        }

9. Program to determine whether a given integer is odd or even

                        #include
                        #include
                        void main()
                        {
                        int number,rem;
                        clrscr();
                        printf(“\n Enter an integer number:”);
                        scanf(“%d”,&number);

                        if(rem==0)
                        printf(“\nThe number %d is even”,number);

11. Program to determine if a year is a leap year or not

                        #include
                        #include
                        void main()
                        {
                        int year,rem_4,rem_100,rem_400;
                        clrscr();
                        printf(“\nEnter the year to be tested:”);
                        scanf(“%d”,&year);
                        rem_4 = year % 4;
                        rem_100=year%100;
                        rem_400=year%400;
                        if(rem_4 == 0 && rem_100 != 0 || rem_400 == 0)
                        printf(“\nThe given year %d is a leap year”,year);
                        else
                        printf(“\nThe given year %d is not a leap year”,year);
                        }
